from abc import ABC, abstractmethod
from urcl.types import Mnemonic, GeneralRegister, StackPointer, BasePointer, Label

class Operand(ABC):
    ...

class Instruction(ABC):
    
    @abstractmethod
    def get_mnemonic(self) -> Mnemonic:
        "Returns the instruction's Mnemonic."
    
    @abstractmethod
    def get_operands(self) -> list[Operand]:
        "Returns the instruction's operands."
    
    @abstractmethod
    def is_simple_calculation(self) -> bool:
        "Returns True if the instruction is a normal calculation."
    
    @abstractmethod
    def get_line_and_column(self) -> tuple[int, int]:
        "Returns line and column number on which this instruction is defined."
    
    @abstractmethod
    def get_destination_register(self) -> GeneralRegister | StackPointer | BasePointer | None:
        "Returns the register to which this instruction writes, if any."
    
    @abstractmethod
    def get_jump_target(self) -> Label | None:
        "Returns place to which this instruction branches, if any."

class Program(ABC):

    @abstractmethod
    def get_min_word_size(self) -> int:
        "Returns the smallest word size for which the program will run, in bits."
    
    @abstractmethod
    def get_max_word_size(self) -> int:
        "Returns the largest word size for which the program will run, in bits."
    
    @abstractmethod
    def get_required_stack_size(self) -> int:
        "Returns the amount of stack space required for the program to run, in words."
    
    @abstractmethod
    def get_required_address_space_size(self) -> int:
        "Returns the size of the smallest address space in which the program will run, in words. Also known as 'minram'"
    
    @abstractmethod
    def get_required_bss_size(self) -> int:
        "Returns the minimum BSS section size that the program needs to run, in words. Also known as 'minheap'"
    
    @abstractmethod
    def is_run_ram(self) -> bool:
        "Checks is the 'RUN' header is 'ROM'."
    
    @abstractmethod
    def is_run_rom(self) -> bool:
        "Checks is the 'RUN' header is 'RAM'."
    
    @abstractmethod
    def get_required_register_count(self) -> int:
        "Returns the minimum amount of general-purpose registers that the program needs to run."
    
    @abstractmethod
    def get_defined_words(self) -> list[int]:
        "Returns a list of words DW'd in the program."
    
    @abstractmethod
    def get_instructions(self) -> list[Instruction]:
        "Returns the programs instructions, excluding labels."
    
    @abstractmethod
    def get_code(self) -> list[Instruction | str]:
        "Returns the programs instructions, including labels."
    
    @abstractmethod
    def get_macro(self, name: str) -> str | None:
        "Returns the value of the supplied macro, or None if it is undefined."
    
    @abstractmethod
    def label_is_executable(self, name: str) -> bool:
        "Returns True if the supplied label points to executable code."
    
    @abstractmethod
    def get_defined_labels(self) -> list[str]:
        "Returns a list of labels defined by the program."
    
    @abstractmethod
    def get_referenced_labels(self) -> list[str]:
        "Returns a list of labels referenced in the program."
    
    @abstractmethod
    def get_undefined_labels(self) -> list[str]:
        "Returns a list of unresolved label references in the program."
    
    @abstractmethod
    def get_defined_macros(self) -> list[str]:
        "Returns a list of macros defined by the program."
    
    @abstractmethod
    def get_referenced_macros(self) -> list[str]:
        "Returns a list of macros referenced in the program."
    
    @abstractmethod
    def get_undefined_macros(self) -> list[str]:
        "Returns a list of unresolved macro references in the program."
    
    @abstractmethod
    def get_source(self) -> str:
        "Returns the program's source code."
    
    @abstractmethod
    def print(self, name: str) -> str:
        "Returns the program as a string."
