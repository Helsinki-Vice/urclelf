"This module takes a stream of URCL tokens and parses it into a concrete syntax tree"

import enum

from dataclasses import dataclass
from urcl.types import Mnemonic, Label, RelativeAddress, Character, Port, GeneralRegister, DefinedImmediate, BasePointer, StackPointer, Header
from urcl.lex import TokenType, TokenStream, Token
from error import Traceback

class CSTNodeType(enum.Enum):
    PROGRAM = "INSTRUCTION"
    LINE = "LINE"

    INSTRUCTION = "INSTRUCTION"
    MACRO_DEFINITION = "INSTRUCTION"
    HEADER = "INSTRUCTION"
    LABEL = "LABEL"

    MNEMONIC = "MNEMONIC"
    OPERAND = "OPERAND"

    DEFINE_KEYWORD = "MNEMONIC"
    DEFINED_IDENTIFIER = "OPERAND"
    DEFINITION = "OPERAND"


class CSTNode:
    source: str
    line_number: int
    column_number: int
